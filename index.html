<script>

const jobs = [

  "葬儀屋", "宝石商", "私設探偵", "宮廷料理人",

  "密輸業者", "修道士", "人形師", "薬師",

  "逃がし屋", "地図職人"

];

const personalities = [

  "享楽主義", "義理堅い", "虚無的", "世話焼き",

  "他人に興味がない", "嘘がつけない", "承認欲求が強い",

  "冷酷で合理的", "ロマンチスト", "自己犠牲型"

];

const secrets = [

  "人を殺したことがある",

  "本名を捨てている",

  "監視されている",

  "不老である",

  "触れたものの記憶が見える",

  "誰かに成り代わっている",

  "致命的な病を隠している",

  "大切な人を裏切っている",

  "過去を覚えていない",

  "自分の死期を知っている"

];

function rand(arr) {

  return arr[Math.floor(Math.random() * arr.length)];

}

function draw() {

  const job = rand(jobs);

  const personality = rand(personalities);

  const secret = rand(secrets);

  document.getElementById("result").innerHTML = `

    職業：${job}<br>

    性格：${personality}<br>

    秘密：${secret}

  `;

}

</script>
